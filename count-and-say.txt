class Solution {public:    string cast(string in)	{		int count = 1;		char ch = in[0];		string out;				for (int i = 1; i <= in.size(); i++)		{			if (in[i] == ch) {count++;}			else			{				char chr = count + '0';				out = out + chr;				out = out + ch;				ch = in[i];				count = 1;			}		}		return out;	}	string countAndSay(int n) {        if (n == 1) return "1";		string str = "1";		string res;				for (int i = 1; i < n; i++)		{			res = cast(str);			str = res;		}		return res;    }};
